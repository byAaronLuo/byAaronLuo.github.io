      1.1.3. 单引号被过滤

  7. $$

    select 'test'
-- 等价于
select $$test$$

  8. $quote$

    -- 适用于当$$ 不能利用的情况下
select $quote$test$quote$

  9. CHR

    SELECT CHR(65)||CHR(66)||CHR(67)||CHR(68)||CHR(69)||CHR(70)||CHR(71)||CHR(72);
-- 等效于
SELECT 'ABCDEFGH';

    注意：您不能同时使用'、$$ 、$quote$，因此，如果您需要转义以单引号开头的字符串，则将无法使用$$(即这种语句是无效的SELECT 'test$$;)
